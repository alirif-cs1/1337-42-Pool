#include "ft_list.h"
#include <stdio.h>

void	ft_print_adresses(t_list *e2)
{
	while (e2->next != NULL)
	{
		printf("Ajo nchofo %p ==> next dyalha ==> %p\n", e2, e2->next);
		e2 = e2->next;
	}
	printf("Ajo nchofo %p ==> next dyalha ==> %p\n", e2, e2->next);
}

void	ft_print_all(t_list *e2)
{
	while (e2->next != NULL)
	{
		printf("%s | ", e2->data);
		e2 = e2->next;
	}
	printf("%s\n", e2->data);
}

void	ft_list_sort(t_list **begin_list, int (*cmp)())
{
	t_list	*first;
	t_list	*tmp;
	t_list	*last;
	int		edited;
	int		is_first;

	edited = 1;
	first = begin_list[0];
	while (edited == 1)
	{
		last = NULL;
		is_first = 1;
		edited = 0;
		begin_list[0] = first;
		printf("Count \n");
		while (begin_list[0]->next != NULL)
		{
			ft_print_adresses(first);
			ft_print_all(first);
			printf("=========Last : %p==========\n", last);
			printf("Comparaison %s <=> %s \n", begin_list[0]->data, begin_list[0]->next->data);
			if ((*cmp)(begin_list[0]->data, begin_list[0]->next->data) > 0)
			{
				printf("Ok dkhol\n");
				if (last != NULL)
					last->next = begin_list[0]->next;
				tmp = begin_list[0]->next->next;
				begin_list[0]->next->next = begin_list[0];
				if (is_first == 1)
					first = begin_list[0]->next;
				last = begin_list[0]->next;
				begin_list[0]->next = tmp;
				edited = 1;
			}
			else
			{
				last = begin_list[0];
				begin_list[0] = begin_list[0]->next;
			}
			is_first = 0;
			ft_print_adresses(first);
			ft_print_all(first);
			getchar();
		}
	}
	begin_list[0] = first;
}
